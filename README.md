    # Inicializar Git
    git init

    # Configurar remoto
    git remote add origin <link copiado do github>

    # Criar/editar .gitignore

    # Adicionar ficheiros e fazer commit
    git add .
    git commit -m "Primeiro commit"

    # Definir branch principal
    git branch -M main

    # Enviar para o remoto
    git push -u origin main

    # Fluxo diário
    git add .  # Novos ficheiros
    git commit -m "mensagem"  # Ficheiros alterados
    git push # Finaliza
